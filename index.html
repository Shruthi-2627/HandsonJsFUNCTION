Question 1 Create one function with zero parameter having a console statement.
Solution 
        function Hello(){ //created a function with no parameters, named as Hello()
            console.log("Hello Shruthi");
        }
        
        //function call from main program
        Hello(); // Hello Shruthi
Question 2 Create one function which takes two values as a parameter and print the sum of them as "Sum of 3, 4 is 7".
Solution 
        function Add(num1, num2){ //created a function with two parameters, named as Add()
            console.log(`Sum of ${num1} and ${num2} is ${num1+num2}`);
        }
        
        //function call from main program
        Add(3, 4); // Sum of 3 and 4 is 7

Question 7 Write a function that accepts parameter n and returns factorial of n
Solution 
        Factorial=(n)=>{
            let factorial=1;
            for(let i=1; i<=n; i++)
            {
            factorial*= i;
            }
            return factorial;
        }
        console.log(Factorial(10));

        
output


        3628800
